    **Step 1: Create API client helper**

    Create `apps/web/src/lib/api.ts`:
    - `fetchApi(path, options?)` — a server-side helper that reads the access_token from cookies and adds it as Bearer token header. Used by all server actions and server components to call NestJS API.
    ```typescript
    import { cookies } from 'next/headers';

    export async function fetchApi(path: string, options: RequestInit = {}) {
      const cookieStore = await cookies();
      const token = cookieStore.get('access_token')?.value;
      const res = await fetch(`${process.env.API_URL}${path}`, {
        ...options,
        headers: {
          'Content-Type': 'application/json',
          ...(token ? { Authorization: `Bearer ${token}` } : {}),
          ...options.headers,
        },
      });
      if (!res.ok) {
        const error = await res.json().catch(() => ({ message: 'Request failed' }));
        throw new Error(error.message || `API error: ${res.status}`);
      }
      return res.json();
    }
    ```
